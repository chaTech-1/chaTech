DROP TABLE IF EXISTS users;

CREATE TABLE users(
user_id SERIAL PRIMARY KEY,
email VARCHAR(225),
name VARCHAR(225));

CREATE TABLE topics(
topic_id SERIAL PRIMARY KEY,
name VARCHAR(225));


CREATE TABLE messages(
message_id INT PRIMARY KEY,
time  DATETIME,
message TEXT,
chat_id FOREIGN KEY,
user_id FOREIGN KEY,
REFERENCES users(user_id)
);


CREATE TABLE chats(
chat_id INT PRIMARY KEY,
user_id FOREIGN KEY);
